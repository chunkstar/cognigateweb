Quick Start

Get up and running in under 5 minutes

// Install via npm
npm install cognigate

// Create gateway with all features enabled
import { createGateway } from 'cognigate';

const ai = createGateway({
  dailyBudget: 10.00,           // $10 daily limit
  cacheEnabled: true,          // Save money with caching
  semanticCaching: true,      // Match similar prompts
  compressionLevel: 'high',    // 40% token reduction
  cloudProviders: {
    openai: {
      apiKey: process.env.OPENAI_API_KEY
    },
    google: {
      apiKey: process.env.GOOGLE_API_KEY
    }
  },
  localFallback: {
    enabled: true,           // Free fallback to Ollama
    providers: ['ollama']
  },
  webhooks: {
    slack: process.env.SLACK_WEBHOOK  // Get budget alerts
  }
});

// Use it!
const answer = await ai.complete('Explain TypeScript in one sentence');
console.log(answer);

// Track spending in real-time
const status = ai.getBudgetStatus();
console.log(`Spent: $${status.used.toFixed(4)}`);
console.log(`Remaining: $${status.remaining.toFixed(2)}`);
